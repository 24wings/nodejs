function readImage() {
    var inputEl = document.createElement("input");
    inputEl.type = "file";
    inputEl.onchange = function() {
        var file = inputEl.files[0];
        if (file) {
            var reader = new FileReader();
            reader.onload = function(event) {
                var base64 = event.target.result;

            }
            reader.readAsDataURL(file);
        }

    }
}
